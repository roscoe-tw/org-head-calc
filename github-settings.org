** publish.yml
#+begin_src yaml :tangle .github/workflows/publish.yml
  name: Publish HTML Slides

  on:
    push:
      branches: [ main ]

  env:
    # 提前切換 Node 24（避免未來壞掉）
    FORCE_JAVASCRIPT_ACTIONS_TO_NODE24: true

  jobs:
    publish:
      runs-on: ubuntu-latest

      permissions:
        contents: write

      steps:
        # 🔴 關鍵：完全禁止 submodule 干擾
        - name: Disable submodule completely
          run: |
            git config --global submodule.recurse false

        # 🔴 關鍵：乾淨 checkout
        - name: Checkout code
          uses: actions/checkout@v4
          with:
            submodules: false
            fetch-depth: 0
            clean: true

        # 🧰 安裝環境
        - name: Install Emacs & Maxima
          run: |
            sudo apt-get update
            sudo apt-get install -y emacs-nox maxima

        # 📄 匯出 HTML（reveal.js 走 CDN）
        - name: Generate HTML Slides
          run: |
            mkdir -p ~/.emacs.d
            emacs --batch -l publish.el calculation.org -f org-reveal-export-to-html

        # 🚀 部署到 gh-pages
        - name: Deploy to GitHub Pages
          uses: peaceiris/actions-gh-pages@v3
          with:
            github_token: ${{ secrets.GITHUB_TOKEN }}
            publish_dir: ./
            publish_branch: gh-pages
            enable_jekyll: false
#+end_src
