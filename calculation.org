** 最小需求厚度 (t)
公式如下：
\[ t = \frac{P \cdot L \cdot M}{2SE - 0.2P} \]

#+NAME: calc_t
#+BEGIN_SRC maxima :var params=design_params :exports results :results html :eval yes
/* 1. 輔助函數：強制去空白並轉換字串比對 */
find_val_sym(target, tbl) := block(
    [val: false],
    for row in tbl do (
        if ssubst("", " ", string(row[2])) = target then val: row[3]
    ),
    if val = false then error("NotFound: ", target) else val
)$

/* 2. 提取所有必要參數 */
P : float(find_val_sym("P", params))$
L : float(find_val_sym("L", params))$
r : float(find_val_sym("r", params))$
S : float(find_val_sym("S", params))$
E : float(find_val_sym("E", params))$

/* 3. 計算邏輯 */
if (P <= 0 or S <= 0 or E <= 0) then error("P, S, E must > 0")$

/* 計算係數 M */
M_val : 0.25 * (3 + sqrt(L/r))$

/* 計算需求厚度 t_req (ASME Sec. VIII Div.1 Appendix 1-4) */
t_req : (P * L * M_val) / (2*S*E - 0.2*P)$

/* 4. 輸出 HTML 結果 */
print(concat("<div id='result-t' class='results-box'>",
             "<strong>(2) 需求厚度計算結果</strong><br/>",
             "當 P = ", string(P), " MPa, S = ", string(S), " MPa, E = ", string(E), "<br/>",
             "計算得 t_req = <strong>", string(t_req), "</strong> mm</div>"))$
#+END_SRC
